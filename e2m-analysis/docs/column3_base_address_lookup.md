## Usage Example

```typescript
import { loadMappingsFromDatabase, calculateAddress } from "./utils/address-calculator.js";

async function example() {
  const mappings = await loadMappingsFromDatabase();

  // Calculate address for a parameter
  const result = calculateAddress("106B", "20", mappings, "EXAMPLE_PARAM");

  console.log(result.hexAddress);     // e.g., "0x010001A8"
  console.log(result.memoryZone);     // e.g., "EEPROM"
  console.log(result.firmwareRegion); // e.g., "EEPROM_DataPlate"
}
```
